function timeToLocalDateString(tm) {
    "use strict";
    let d = new Date(tm);
    let p1 = d.toLocaleTimeString().split(' ');
    let p2 = p1[0].split(':');
    p2 = p2.slice(0, 2);
 
    return d.toLocaleDateString() + '<br>' + p2.join(':') + ' ' + p1[1];
}
